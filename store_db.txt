-- =============================================
-- PostgreSQL Schema for your STORE_ID database
-- =============================================

CREATE TABLE company_branch (
    branch_id integer GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY
);

CREATE TABLE customer (
    customer_id integer GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY
);

CREATE TABLE dept (
    dept_id integer GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    dept_name varchar(50) NOT NULL
);

CREATE TABLE emp_state (
    emp_id integer GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name varchar(50),
    img varchar(4000),
    dept_id varchar(50),
    national_id varchar(50)
);

CREATE TABLE login_emp (
    emp_id integer PRIMARY KEY,
    password varchar(50)
);

CREATE TABLE offers (
    offer_id integer GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY
);

CREATE TABLE offers_pi (
    offer_id integer NOT NULL,
    pi_id integer NOT NULL,
    PRIMARY KEY (offer_id, pi_id)
);

CREATE TABLE personal_info (
    id integer GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name varchar(50) NOT NULL,
    gender char(10) NOT NULL,
    birth_date date NOT NULL,
    user_id integer NOT NULL,
    role_cat_id integer NOT NULL,
    emp_id integer NOT NULL,
    customer_id integer NOT NULL,
    dept_id integer NOT NULL,
    branch_id integer NOT NULL
);

CREATE TABLE product (
    product_id integer GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY
);

CREATE TABLE product_branch (
    product_id integer NOT NULL,
    branch_id integer NOT NULL,
    PRIMARY KEY (product_id, branch_id)
);

CREATE TABLE role (
    role_cat_id integer GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name_of_cat varchar(50) NOT NULL
);

CREATE TABLE storehouse (
    storehouse_id integer GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY
);

CREATE TABLE storehouse_branch (
    storehouse_id integer NOT NULL,
    branch_id integer NOT NULL,
    PRIMARY KEY (storehouse_id, branch_id)
);

CREATE TABLE storehouse_product (
    product_id integer NOT NULL,        -- fixed typo from original
    storehouse_id integer NOT NULL,
    PRIMARY KEY (product_id, storehouse_id)
);

CREATE TABLE users (
    user_id integer GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY
);

-- ====================== FOREIGN KEYS ======================
ALTER TABLE offers_pi ADD CONSTRAINT fk_offers_pi_offers 
    FOREIGN KEY (offer_id) REFERENCES offers(offer_id);

ALTER TABLE offers_pi ADD CONSTRAINT fk_offers_pi_personal_info 
    FOREIGN KEY (pi_id) REFERENCES personal_info(id);

ALTER TABLE personal_info ADD CONSTRAINT fk_personal_info_company_branch 
    FOREIGN KEY (branch_id) REFERENCES company_branch(branch_id);

ALTER TABLE personal_info ADD CONSTRAINT fk_personal_info_customer 
    FOREIGN KEY (customer_id) REFERENCES customer(customer_id);

ALTER TABLE personal_info ADD CONSTRAINT fk_personal_info_dept 
    FOREIGN KEY (dept_id) REFERENCES dept(dept_id);

ALTER TABLE personal_info ADD CONSTRAINT fk_personal_info_emp_state 
    FOREIGN KEY (emp_id) REFERENCES emp_state(emp_id);

ALTER TABLE personal_info ADD CONSTRAINT fk_personal_info_role 
    FOREIGN KEY (role_cat_id) REFERENCES role(role_cat_id);

ALTER TABLE personal_info ADD CONSTRAINT fk_personal_info_users 
    FOREIGN KEY (user_id) REFERENCES users(user_id);

ALTER TABLE product_branch ADD CONSTRAINT fk_product_branch_company_branch 
    FOREIGN KEY (branch_id) REFERENCES company_branch(branch_id);

ALTER TABLE product_branch ADD CONSTRAINT fk_product_branch_product 
    FOREIGN KEY (product_id) REFERENCES product(product_id);

ALTER TABLE storehouse_branch ADD CONSTRAINT fk_storehouse_branch_company_branch 
    FOREIGN KEY (branch_id) REFERENCES company_branch(branch_id);

ALTER TABLE storehouse_branch ADD CONSTRAINT fk_storehouse_branch_storehouse 
    FOREIGN KEY (storehouse_id) REFERENCES storehouse(storehouse_id);

ALTER TABLE storehouse_product ADD CONSTRAINT fk_storehouse_product_product 
    FOREIGN KEY (product_id) REFERENCES product(product_id);

ALTER TABLE storehouse_product ADD CONSTRAINT fk_storehouse_product_storehouse 
    FOREIGN KEY (storehouse_id) REFERENCES storehouse(storehouse_id);

-- ====================== INDEXES (optional but kept) ======================
CREATE UNIQUE INDEX ix_emp_state ON emp_state(emp_id);
CREATE INDEX ix_personal_info ON personal_info(id);
